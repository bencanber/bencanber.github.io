JAVA_Socket通信实例

发表于 2018-01-11

在网络通信过程中，主要有两种传输协议TCP（传输控制协议）和UDP（数据报传输协议），TCP是一种面向连接的传输协议，具有需要建立连接通道、数据无限制、速度慢、可靠等特点。UDP是一种面向无连接的传输协议，具有数据打包传输、数据有限制、速度快、不可靠等特点。接下来主要针对这两种协议的java实现进行详细描述及代码实现。

网络编程三要素（IP、PORT、协议）

###TCP：

客户端：
    1、    创建发送端的socket对象，这一步建立成功说明建立连接成功
    2、    获取输出流，写数据
    3、    获取输入流，读取服务器返回数据
    4、    释放资源

服务器端：
    1、    创建接收端Serversocket对象，监听客户端连接
    2、    一旦接收到客户端连接，即返回一个socket对象
    3、    获取输入流，读数据
    4、    获取输出流，写数据
    5、    释放资源

###UDP：
客户端：
1、 创建发送端socket对象
2、 创建数据并进行打包
3、 发送数据
4、 释放资源
服务器端：
1、 创建接收端socket对象
2、 建立一个缓冲区接收数据
3、 接收数据并解析
4、 释放资源

import java.io.IOException;
import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;

public class Send {
    public static void main(String[] args) throws IOException {
        //创建发送端socket
        DatagramSocket ds = new DatagramSocket();

        //创建数据并打包
        byte[] bys = "hello server".getBytes();

        DatagramPacket dp = new DatagramPacket(bys,bys.length,
                InetAddress.getByName("127.0.0.1"),12345);
        //发送数据包
        ds.send(dp);
        //释放资源
        ds.close();
    }
}、、、

import java.io.IOException;
import java.net.DatagramPacket;
import java.net.DatagramSocket;

public class Receive {
    public static void main(String[] args) throws IOException {
        //创建接收端socket对象
        DatagramSocket ds = new DatagramSocket(12345);
        //创建缓冲区
        byte[] bys = new byte[2014];
        DatagramPacket dp = new DatagramPacket(bys,bys.length);
        //接受数据包
        ds.receive(dp);
        //解析数据
        String data = new String(dp.getData(),0,dp.getLength());
        //获取来源ip
        String ip = dp.getAddress().getHostAddress();

        System.out.println(ip+":"+data);
        //释放资源
        ds.close();

    }
}、、、
